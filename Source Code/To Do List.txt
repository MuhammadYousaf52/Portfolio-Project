    <script>
      const button = document.getElementById("add-task");
      const input = document.getElementById("task");
      button.addEventListener("click", () => {
        const value = input.value;
        const trimmedValue = value.trim();
        if (trimmedValue == "") {
          alert("You didn't type anything");
        } else {
          const taskadded = document.querySelector(".added-tasks");
          const taskdiv = document.createElement("div");
          taskdiv.className = "task";
          taskdiv.innerHTML = `<div class="text">${trimmedValue}</div>
                <img class="delete" src="icons/trash.svg" alt="delete">`;
          taskadded.appendChild(taskdiv);
          input.value = "";

          const trash = taskdiv.querySelector(".delete");
          trash.addEventListener("click", function(){
            this.parentElement.remove()
          });
        }
      });
      input.addEventListener("keydown", function (e) {
        if(e.key == "Enter"){
            button.click();
        }
      })
    </script>
